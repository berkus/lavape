function Exception::Show

declare
   String callStack
do
   set callStack <== Exception::CallStack;
   if
      self.message = Ø
   then
      if
         callStack = Ø
      then
         call self.code.String.MessageBox
      else
         call ( self.code.String + "\n\n" + Exception::CallStack ).MessageBox
      #if
   else
      if
         callStack = Ø
      then
         call ( self.code.String + ":\n" + self.message ).MessageBox
      else
         call ( self.code.String + ":\n" + self.message + "\n\n" + Exception::CallStack ).MessageBox
      #if
   #if
#declare
